55      @Test
56      public void testConfigureComponent_1() throws Exception {
57          ConverterLookup converterLookup = mock(ConverterLookup.class);
58          Field modifiersField = configurator.getClass().getSuperclass().getSuperclass().getSuperclass().getDeclaredField("converterLookup");
59          modifiersField.setAccessible(true);
60          modifiersField.set(configurator, converterLookup);
61          Object component = mock(Object.class);
62          PlexusConfiguration configuration = mock(PlexusConfiguration.class);
63          ExpressionEvaluator expressionEvaluator = mock(ExpressionEvaluator.class);
64          org.codehaus.plexus.classworlds.realm.ClassRealm containerRealm = mock(org.codehaus.plexus.classworlds.realm.ClassRealm.class);
65          ConfigurationListener listener = mock(ConfigurationListener.class);
66          doNothing().when(configurator).addDependenciesToClassRealm(any(ExpressionEvaluator.class), any(AbstractIncludeProjectDependenciesComponentConfigurator.Classpath.class), any(org.codehaus.plexus.classworlds.realm.ClassRealm.class));
67  
68          configurator.configureComponent(component, configuration, expressionEvaluator, containerRealm, listener);
69  
70          verify(configurator, atLeastOnce()).addDependenciesToClassRealm(any(ExpressionEvaluator.class), any(AbstractIncludeProjectDependenciesComponentConfigurator.Classpath.class), any(org.codehaus.plexus.classworlds.realm.ClassRealm.class));
71          verify(converterLookup, atLeastOnce()).registerConverter(any(ClassRealmConverter.class));
72      }

94      @Test
95      @SuppressWarnings("deprecation")
96      public void testAddProjectCompileDependenciesToClassRealm() throws Exception {
97          ExpressionEvaluator expressionEvaluator = mock(ExpressionEvaluator.class);
98          List classpathElements = singletonList("CLASSPATH_ELEMENT");
99          doReturn(classpathElements).when(expressionEvaluator).evaluate(anyString());
100         org.codehaus.classworlds.ClassRealm containerRealm = mock(org.codehaus.classworlds.ClassRealm.class);
101         configurator.addDependenciesToClassRealm(expressionEvaluator, IncludeProjectCompileDependenciesComponentConfigurator.Classpath.COMPILE, containerRealm);
102         verify(expressionEvaluator, times(1)).evaluate(anyString());
103         verify(containerRealm, times(1)).addConstituent(any(URL.class));
104     }
