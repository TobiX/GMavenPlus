39  /**
40   * Launches a Groovy console window bound to the current project.
41   * Note that this mojo requires Groovy >= 1.5.0.
42   * Note that it references the plugin classloader to pull in dependencies
43   * Groovy didn't include (for things like Ant for AntBuilder, Ivy for @grab,
44   * and Jansi for Groovysh).
45   *
46   * @author Keegan Witt
47   * @since 1.1
48   */
49  @Mojo(name="console", requiresDependencyResolution=ResolutionScope.TEST, configurator="include-project-test-dependencies")
50  public class ConsoleMojo extends AbstractToolsMojo {
51  
52      /**
53       * Executes this mojo.
54       *
55       * @throws MojoExecutionException If an unexpected problem occurs. Throwing this exception causes a "BUILD ERROR" message to be displayed
56       * @throws MojoFailureException If an expected problem (such as an invocation failure) occurs. Throwing this exception causes a "BUILD FAILURE" message to be displayed
57       */
58      @Override
59      public void execute() throws MojoExecutionException, MojoFailureException {
60          classWrangler = new ClassWrangler(Thread.currentThread().getContextClassLoader(), getLog());
61  
62          logPluginClasspath();
63          classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
64  
65          try {
66              getLog().debug("Project test classpath:\n" + project.getTestClasspathElements());
67          } catch (DependencyResolutionRequiredException e) {
68              getLog().warn("Unable to log project test classpath", e);
69          }
70  
71          if (groovyVersionSupportsAction()) {
72              final SecurityManager sm = System.getSecurityManager();
73              try {
74                  if (!allowSystemExits) {
75                      System.setSecurityManager(new NoExitSecurityManager());
76                  }
77  
78                  // get classes we need with reflection
79                  Class<?> consoleClass = classWrangler.getClass("groovy.ui.Console");
80                  Class<?> bindingClass = classWrangler.getClass("groovy.lang.Binding");
81  
82                  // create console to run
83                  Object console = setupConsole(consoleClass, bindingClass);
84  
85                  // run the console
86                  invokeMethod(findMethod(consoleClass, "run"), console);
87  
88                  // TODO: for some reason instantiating AntBuilder before calling run() causes its stdout and stderr streams to not be captured by the Console
89                  bindAntBuilder(consoleClass, bindingClass, console);
90  
91                  // wait for console to be closed
92                  Set<Thread> threadSet = Thread.getAllStackTraces().keySet();
93                  Thread[] threadArray = threadSet.toArray(new Thread[threadSet.size()]);
94                  Thread consoleThread = null;
95                  for (Thread thread : threadArray) {
96                      if ("AWT-Shutdown".equals(thread.getName())) {
97                          consoleThread = thread;
98                          break;
99                      }
100                 }
101                 if (consoleThread != null) {
102                     try {
103                         consoleThread.join();
104                     } catch (InterruptedException e) {
105                         throw new MojoExecutionException("Mojo interrupted while waiting for Console thread to end.", e);
106                     }
107                 } else {
108                     throw new MojoFailureException("Unable to locate Console thread to wait on.");
109                 }
110             } catch (ClassNotFoundException e) {
111                 throw new MojoExecutionException("Unable to get a Groovy class from classpath.  Do you have Groovy as a compile dependency in your project or the plugin?", e);
112             } catch (InvocationTargetException e) {
113                 if (e.getCause() instanceof NoClassDefFoundError && "org/apache/ivy/core/report/ResolveReport".equals(e.getCause().getMessage())) {
114                     throw new MojoExecutionException("Groovy 1.7.6 and 1.7.7 have a dependency on Ivy to run the console.  Either change your Groovy version or add Ivy as a project or plugin dependency.", e);
115                 } else {
116                     throw new MojoExecutionException("Error occurred while calling a method on a Groovy class from classpath.", e);
117                 }
118             } catch (IllegalAccessException e) {
119                 throw new MojoExecutionException("Unable to access a method on a Groovy class from classpath.", e);
120             } catch (InstantiationException e) {
121                 throw new MojoExecutionException("Error occurred while instantiating a Groovy class from classpath.", e);
122             } finally {
123                 if (!allowSystemExits) {
124                     System.setSecurityManager(sm);
125                 }
126             }
127         } else {
128             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support running a console.  The minimum version of Groovy required is " + minGroovyVersion + ".  Skipping console startup.");
129         }
130     }

132     /**
133      * Binds a new AntBuilder to the project properties.
134      *
135      * @param consoleClass the groovy.ui.Console class to use
136      * @param bindingClass the groovy.lang.Binding class to use
137      * @param console the groovy.ui.Console object to use
138      * @throws ClassNotFoundException
139      * @throws IllegalAccessException
140      * @throws InvocationTargetException
141      * @throws InstantiationException
142      */
143     protected void bindAntBuilder(Class<?> consoleClass, Class<?> bindingClass, Object console) throws ClassNotFoundException, IllegalAccessException, InvocationTargetException, InstantiationException {
144         if (properties.containsKey("ant")) {
145             Class<?> groovyShellClass = classWrangler.getClass("groovy.lang.GroovyShell");
146             Object shell = getField(findField(consoleClass, "shell", groovyShellClass), console);
147             Object binding = invokeMethod(findMethod(groovyShellClass, "getContext"), shell);
148             Object antBuilder = invokeConstructor(findConstructor(classWrangler.getClass("groovy.util.AntBuilder")));
149             if (bindPropertiesToSeparateVariables) {
150                 invokeMethod(findMethod(bindingClass, "setVariable", String.class, Object.class), binding, "ant", antBuilder);
151             } else {
152                 properties.put("ant", antBuilder);
153             }
154         }
155     }
156 
157     /**
158      * Instantiates a Groovy Console.
159      *
160      * @param consoleClass the Console class
161      * @param bindingClass the Binding class
162      * @return the instantiated Console
163      * @throws InstantiationException when a class needed for creating a console cannot be instantiated
164      * @throws IllegalAccessException when a method needed for creating a console cannot be accessed
165      * @throws InvocationTargetException when a reflection invocation needed for creating a console cannot be completed
166      */
167     protected Object setupConsole(final Class<?> consoleClass, final Class<?> bindingClass) throws InvocationTargetException, IllegalAccessException, InstantiationException {
168         Object binding = invokeConstructor(findConstructor(bindingClass));
169         initializeProperties();
170         Method setVariable = findMethod(bindingClass, "setVariable", String.class, Object.class);
171         if (bindPropertiesToSeparateVariables) {
172             for (Object k : properties.keySet()) {
173                 invokeMethod(setVariable, binding, k, properties.get(k));
174             }
175         } else {
176             invokeMethod(setVariable, binding, "properties", properties);
177         }
178 
179         return invokeConstructor(findConstructor(consoleClass, ClassLoader.class, bindingClass), Thread.currentThread().getContextClassLoader(), binding);
180     }
