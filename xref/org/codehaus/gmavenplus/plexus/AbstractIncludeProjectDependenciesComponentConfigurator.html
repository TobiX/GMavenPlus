34  /**
35   * A custom ComponentConfigurator which adds the project's classpath elements
36   * to the plugin's ClassRealm.
37   * Note that there are two method signatures for addDependenciesToClassRealm,
38   * since different versions of Maven have different expectations.
39   *
40   * @author <a href="http://stackoverflow.com/a/2659324/160256">Brian Jackson</a>
41   * @author Keegan Witt
42   * @since 1.0-beta-3
43   */
44  public abstract class AbstractIncludeProjectDependenciesComponentConfigurator extends AbstractComponentConfigurator {
45  
46  //    private static final Log LOG = new SystemStreamLog();
47  
48      /**
49       * Adds the project's compile dependencies to the specified ClassRealm.
50       *
51       * @param expressionEvaluator The expression evaluator to use to get project elements
52       * @param classpath The classpath to load into the container realm
53       * @param containerRealm The ClassRealm to add dependencies to
54       * @throws ComponentConfigurationException when parsing components configuration fails
55       */
56      protected void addDependenciesToClassRealm(final ExpressionEvaluator expressionEvaluator, final Classpath classpath, final org.codehaus.plexus.classworlds.realm.ClassRealm containerRealm) throws ComponentConfigurationException {
57          List classpathElements;
58  
59          try {
60              classpathElements = (List) expressionEvaluator.evaluate("${project." + classpath.toString().toLowerCase() + "ClasspathElements}");
61          } catch (ExpressionEvaluationException e) {
62              throw new ComponentConfigurationException("There was a problem evaluating: ${project." + classpath.toString().toLowerCase() + "ClasspathElements}.", e);
63          }
64  
65          // add the project dependencies to the ClassRealm
66          final URL[] urls = buildURLs(classpathElements);
67          for (URL url : urls) {
68              containerRealm.addURL(url);
69          }
70      }
71  
72      /**
73       * Adds the project's compile dependencies to the specified ClassRealm.
74       *
75       * @param expressionEvaluator The expression evaluator to use to get project elements
76       * @param classpath The classpath to load into the container realm
77       * @param containerRealm The ClassRealm to add dependencies to
78       * @throws ComponentConfigurationException when parsing components configuration fails
79       */
80      @SuppressWarnings("deprecation")
81      protected void addDependenciesToClassRealm(final ExpressionEvaluator expressionEvaluator, final Classpath classpath, final org.codehaus.classworlds.ClassRealm containerRealm) throws ComponentConfigurationException {
82          List classpathElements;
83  
84          try {
85              classpathElements = (List) expressionEvaluator.evaluate("${project." + classpath.toString().toLowerCase() + "ClasspathElements}");
86          } catch (ExpressionEvaluationException e) {
87              throw new ComponentConfigurationException("There was a problem evaluating: ${project." + classpath.toString().toLowerCase() + "ClasspathElements}.", e);
88          }
89  
90          // add the project dependencies to the ClassRealm
91          final URL[] urls = buildURLs(classpathElements);
92          for (URL url : urls) {
93              containerRealm.addConstituent(url);
94          }
95      }
96  
97      /**
98       * Create an array of URLs for all the elements in the classpath.
99       *
100      * @param classpathElements The classpath elements to create URLs for
101      * @return URLs for all the classpath elements
102      * @throws ComponentConfigurationException when parsing components configuration fails
103      */
104     protected URL[] buildURLs(final List classpathElements) throws ComponentConfigurationException {
105         List<URL> urls = new ArrayList<URL>(classpathElements.size());
106         for (Object element : classpathElements) {
107             try {
108                 final URL url = new File((String) element).toURI().toURL();
109                 if (!urls.contains(url)) {
110                     urls.add(url);
111                 }
112                 // commented out because debug seems to be on all the time
113 //                LOG.debug("Added to project class loader: " + url);
114             } catch (MalformedURLException e) {
115                 throw new ComponentConfigurationException("Unable to access project dependency: " + element + ".", e);
116             }
117         }
118 
119         return urls.toArray(new URL[urls.size()]);
120     }
121 
122     /**
123      * Enum of the various possible classpaths.
124      */
125     public enum Classpath {
126 
127         /**
128          * Compile classpath.
129          */
130         COMPILE,
131 
132         /**
133          * Runtime classpath.
134          */
135         RUNTIME,
136 
137         /**
138          * Test classpath.
139          */
140         TEST,
141 
142         /**
143          * System classpath.
144          */
145         SYSTEM
146     }
